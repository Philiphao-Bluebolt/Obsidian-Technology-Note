给定一个班级所有科目的平均分，判断班级能够被评为优秀班级。评为优秀班级的条件是，所有科目平均分的中位数、平均值、去极平均值**其中之一**高于要求。

输入格式：第一行表示测试样例数。对于每一个样例，第一行表示科目数（不小于3），第二行是每个科目的班平均成绩，第三行三个数字分别是优秀班级中位数、平均值、去极平均值的达标要求。

```
3
4
90 50 76 80
70 80 75
5
90 94 70 60 65
82 85 80
3
98 88 68
70 70 75
```

输出格式如下

```
Yes
No
Yes
```


---
## 代码

```cpp
#include <iostream>
#include <vector>
#include <fstream>
#include <algorithm>
using namespace std;

int main(){
    freopen("./input.txt", "r", stdin);

    int n_test;
    int n_subject;
    int temp;
    int th_med, th_mean, th_mean_np;
    int med, mean, mean_np;
    vector<int> scores;
    
    cin >> n_test;

    for (int i = 0; i < n_test; i++){
     
        // Read test data
        scores.clear();
        cin >> n_subject;
        for (int j = 0; j < n_subject; j++){
            cin >> temp;
            scores.push_back(temp);
        }
        cin >> th_med >> th_mean >> th_mean_np;
        sort(scores.begin(), scores.end());

        // Calculate the median
        med = (n_subject % 2 == 0) ? (scores[n_subject/2]+scores[n_subject/2 - 1]) : scores[(n_subject-1)/2];
        if (med >= th_med){
            cout << "Yes" << "\n";
            continue;
        }

        // Calculate the mean and the mean without peak
        mean = 0;
        mean_np = 0;
        for (int i = 0; i < n_subject; i++){
            mean += scores[i];
            if (i != 0 && i != n_subject - 1) mean_np += scores[i];
        }
        mean /= n_subject;
        mean_np /= n_subject-2;

        if (mean >= th_mean || mean_np >= th_mean_np){
            cout << "Yes" << "\n";
            continue;
        }
        else cout << "No" << "\n";
        
    }
}
```
